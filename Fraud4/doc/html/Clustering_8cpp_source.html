279void Clustering::run(){
280
281 //Asignamos clusters aleatorios a las localizaciones.
282 initialClusterAssignment();
283
284 //Inicializamos
285 bool repetir_interaccion = true;
286 _numIterations = 0;
287
288 while(repetir_interaccion)
289 {
290 _numIterations++;
291
292 /*Apartado 1. Vamos a recalcular los centroides. Para ello,
293 acumulamos la suma de coordenadas de cada cluster*/
294
295 //Inicializamos variables.
296 VectorLocation nuevo_centroide(_K);
297 VectorInt contador(_K);
298
299 //Sumamos todas las coordenadas de cada cluster.
300 for(int i = 0; i < _locations.getSize(); i++)
301 {
302 int j = _clusters.at(i);
303 nuevo_centroide.at(j).setX(nuevo_centroide.at(j).getX() + _locations.at(i).getX());
304 nuevo_centroide.at(j).setY(nuevo_centroide.at(j).getY() + _locations.at(i).getY());
305 contador.at(j)++;
306 }
307
308 //Hacenmos la media
309 for(int j = 0; j < _K; j++)
310 {
311 if(contador.at(j) > 0){
312 _centroids.at(j).setX(nuevo_centroide.at(j).getX() / contador.at(j));
313 _centroids.at(j).setY(nuevo_centroide.at(j).getY() / contador.at(j));
314 }
315 }
316
317 /*Aoartado 2. Vamos a reasignar cada localizacion a su centroide mas cercano.
318 Entonces, si algun parametro ha cambiado, el repetir_interaccion se mantiene en true
319 y aumentaremos el _numIterations.*/
320
321 repetir_interaccion = false;
322
323 for(int i = 0; i < _locations.getSize(); i++){
324 int mejor_cluster = 0;
325 double mejor_dist = _locations.at(i).squaredDistance(_centroids.at(0));
326
327 for(int j = 1; j < _K; j++){
328
329 double dist = _locations.at(i).squaredDistance(_centroids.at(j));
330
331 //Si su distancia es menor, se establece como el mejor.
332 if(dist < mejor_dist)
333 {
334 mejor_dist = dist;
335 mejor_cluster = j;
336 }
337 }
338
339 if(_clusters.at(i) != mejor_cluster){
340 _clusters.at(i) = mejor_cluster;
341 repetir_interaccion = true;
342 }
343 }
344 }
345
346 calculateSumWCV();
347 _isDone = true;
348}
