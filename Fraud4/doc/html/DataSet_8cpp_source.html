530DataSet DataSet::getReducedDataSet(Clustering clustering)
531{
532 if (!clustering.isDone())
533 {
534 throw std::invalid_argument("DataSet::getReducedDataSet clustering no ejecutado");
535 }
536 if (clustering.getNumLocations() != _nLocations)
537 {
538 throw std::invalid_argument("DataSet::getReducedDataSet tamanio incompatible");
539 }
540
541 int newLocations = clustering.getK();
542 DataSet reduced(_nInstances, newLocations);
543 reduced._labels = _labels;
544 reduced._locations = clustering.getCentroids();
545
546 for (int i = 0; i < _nInstances; ++i)
547 {
548 for (int j = 0; j < _nLocations; ++j)
549 {

554 }

562int DataSet::nearestInstance(VectorInt instance, bool selected[]) const
563{
564 if (instance.getSize() != _nLocations)
565 {
566 throw std::invalid_argument("DataSet::nearestInstance tamanio incompatible");
567 }
568 if (_nInstances == 0)
569 {
570 return -1;
571 }
572
573 int nearestIndex = -1;
574 double bestDistance = 0.0;
575
576 for (int i = 0; i < _nInstances; ++i)
577 {
578 if (selected != nullptr && selected[i])
579 {
580 continue;
581 }
582
583 double distance = 0.0;
584 for (int j = 0; j < _nLocations; ++j)
585 {
586 double diff = static_cast<double>(_values[i][j] - instance.at(j));
587 distance += diff * diff;
588 }
589
590 if (nearestIndex == -1 || distance < bestDistance)
591 {
592 bestDistance = distance;
593 nearestIndex = i;
594 }
595 }
596
597 return nearestIndex;
598}

682void classify(const DataSet &datasetModel, DataSet &datasetToClassify,
683 int K1, int K2, bool doReductionDimensionality)
684{
685 if (datasetModel.getNumLocations() != datasetToClassify.getNumLocations())
686 {
687 throw std::invalid_argument("classify: tamanio incompatible");
688 }
689 if (doReductionDimensionality && K1 <= 0)
690 {
691 throw std::invalid_argument("classify: K1 invalido");
692 }
693 if (K2 < 1)
694 {
695 throw std::invalid_argument("classify: K2 invalido");
696 }
697 if (K2 > datasetModel.getNumInstances())
698 {
699 throw std::invalid_argument("classify: K2 demasiado grande");
700 }
701
702 DataSet dsM(datasetModel);
703 DataSet dsC(datasetToClassify);
704
705 if (doReductionDimensionality)
706 {
707 Clustering clustering;
708 clustering.set(dsM.getVectorLocation(), K1);
709 clustering.run();
710
711 dsM = dsM.getReducedDataSet(clustering);
712 dsC = dsC.getReducedDataSet(clustering);
713 }
714
715 int numInstances = dsC.getNumInstances();
716 int numLocations = dsC.getNumLocations();
717 int modelInstances = dsM.getNumInstances();
718
719 for (int i = 0; i < numInstances; ++i)
720 {
721 int votes[2] = {0, 0};
722
723 bool *selected = new bool[modelInstances];
724 for (int s = 0; s < modelInstances; ++s)
725 {
726 selected[s] = false;
727 }
728
729 VectorInt instance(numLocations);
730 for (int j = 0; j < numLocations; ++j)
731 {
732 instance.at(j) = dsC.getValue(i, j);
733 }
734
735 for (int k = 0; k < K2; ++k)
736 {
737 int nearest = dsM.nearestInstance(instance, selected);
738 if (nearest < 0)
739 {
740 break;
741 }
742
743 selected[nearest] = true;
744
745 int label = dsM.getLabel(nearest);
746 if (label == 0)
747 {
748 votes[0]++;
749 }
750 else if (label == 1)
751 {
752 votes[1]++;
753 }
754 }
755
756 delete[] selected;
757
758 if (votes[0] >= votes[1])
759 {
760 dsC.setLabel(i, 0);
761 }
762 else
763 {
764 dsC.setLabel(i, 1);
765 }
766 }
767
768 for (int i = 0; i < datasetToClassify.getNumInstances(); ++i)
769 {
770 datasetToClassify.setLabel(i, dsC.getLabel(i));
771 }
