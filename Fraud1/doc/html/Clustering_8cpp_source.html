253void Clustering::run(){
254
255 //Asignamos clusters aleatorios a las localizaciones.
256 initialClusterAssignment();
257
258 //Inicializamos
259 bool repetir_interaccion = true;
260 _numIterations = 0;
261
262 while(repetir_interaccion)
263 {
264 _numIterations++;
265
266 /*Apartado 1. Vamos a recalcular los centroides. Para ello,
267 acumulamos la suma de coordenadas de cada cluster*/
268
269 //Inicializamos variables.
270 VectorLocation nuevo_centroide(_K);
271 VectorInt contador(_K);
272
273 //Sumamos todas las coordenadas de cada cluster.
274 for(int i = 0; i < _locations.getSize(); i++)
275 {
276 int j = _clusters.at(i);
277 nuevo_centroide.at(j).setX(nuevo_centroide.at(j).getX() + _locations.at(i).getX());
278 nuevo_centroide.at(j).setY(nuevo_centroide.at(j).getY() + _locations.at(i).getY());
279 contador.at(j)++;
280 }
281
282 //Hacenmos la media
283 for(int j = 0; j < _K; j++)
284 {
285 if(contador.at(j) > 0)
286 {
287 _centroids.at(j).setX(nuevo_centroide.at(j).getX() / contador.at(j));
288 _centroids.at(j).setY(nuevo_centroide.at(j).getY() / contador.at(j));
289 }
290 }
291
292 /*Aoartado 2. Vamos a reasignar cada localizacion a su centroide mas cercano.
293 Entonces, si algun parametro ha cambiado, el repetir_interaccion se mantiene en true
294 y aumentaremos el _numIterations.*/
295
296 repetir_interaccion = false;
297
298 for(int i = 0; i < _locations.getSize(); i++)
299 {
300 int mejor_cluster = 0;
301 double mejor_dist = _locations.at(i).squaredDistance(_centroids.at(0));
302
303 for(int j = 1; j < _K; j++)
304 {
305 double dist = _locations.at(i).squaredDistance(_centroids.at(j));
306
307 //Si su distancia es menor, se establece como el mejor.
308 if(dist < mejor_dist)
309 {
310 mejor_dist = dist;
311 mejor_cluster = j;
312 }
313 }
314
315 if(_clusters.at(i) != mejor_cluster){
316 _clusters.at(i) = mejor_cluster;
317 repetir_interaccion = true;
318 }
319 }
320 }
321
322 calculateSumWCV();
323 _isDone = true;
324}
